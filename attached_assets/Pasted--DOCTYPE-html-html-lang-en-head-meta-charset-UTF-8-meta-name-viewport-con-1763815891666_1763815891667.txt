    <script>
        // Use window.location.host for Replit deployment, or a specific URL if known
        // Since both FE and BE run on the same Replit deployment, this works well.
        const SOCKET_URL = window.location.host.includes('replit') ? `https://${window.location.host}` : 'http://localhost:3000';
        const socket = io(SOCKET_URL); 

        const contractAddressInput = document.getElementById('contractAddress');
        const abiInput = document.getElementById('abiInput');
        const eventNameInput = document.getElementById('eventName');
        const startListeningBtn = document.getElementById('startListeningBtn');
        const stopListeningBtn = document.getElementById('stopListeningBtn');
        const logOutput = document.getElementById('log-output');
        const statusMessage = document.getElementById('status-message');
        const initialMessage = document.getElementById('initial-message');

        let isListening = false;

        /**
         * Utility function to display messages in the log area.
         */
        function logMessage(text, type, logData = null) {
            if (initialMessage) initialMessage.classList.add('hidden');
            
            const colorClass = {
                'info': 'text-gray-400',
                'success': 'text-green-400',
                'error': 'text-red-400',
                'log': 'bg-gray-700 p-3 rounded-lg border-l-4 border-indigo-500',
            }[type] || 'text-white';

            const logEntry = document.createElement('div');
            logEntry.className = colorClass + ' text-sm';
            
            let content = `<span class="font-mono text-gray-500">[${new Date().toLocaleTimeString()}]</span> ${text}`;

            if (logData) {
                // The logData is already fully processed by the server
                const txLink = `https://sepolia.etherscan.io/tx/${logData.transactionHash}`;
                // We use Sepolia as a common default for the link, but this should ideally match the chain configured in the BE
                const explorerBase = "https://etherscan.io"; // Default for dynamic explorer link
                
                content = `
                    <div class="font-bold text-indigo-400 mb-1">
                        âœ¨ ${logData.name} Event Detected
                    </div>
                    <div class="text-xs text-json">
                        <span class="text-gray-400 font-semibold">Block:</span> 
                        <a href="${explorerBase}/block/${logData.blockNumber}" target="_blank" class="text-blue-400 hover:text-blue-300 transition">${logData.blockNumber}</a>
                    </div>
                    <div class="text-xs text-json">
                        <span class="text-gray-400 font-semibold">Tx Hash:</span> 
                        <a href="${txLink}" target="_blank" class="text-blue-400 hover:text-blue-300 transition">${logData.transactionHash.substring(0, 10)}...</a>
                    </div>
                    <pre class="text-xs mt-1 text-gray-200 text-json overflow-x-auto">${JSON.stringify(logData.args, null, 2)}</pre>
                `;
            }

            logEntry.innerHTML = content;

            // Prepend the new log entry
            if (logOutput.firstChild) {
                logOutput.insertBefore(logEntry, logOutput.firstChild);
            } else {
                logOutput.appendChild(logEntry);
            }
        }

        /**
         * Displays status updates in the dedicated status box.
         */
        function setStatus(message, type) {
            let color = 'bg-yellow-900 border-yellow-700 text-yellow-300';
            if (type === 'success') color = 'bg-green-900 border-green-700 text-green-300';
            if (type === 'error') color = 'bg-red-900 border-red-700 text-red-300';
            if (type === 'info') color = 'bg-gray-700 border-gray-600 text-gray-400';

            statusMessage.textContent = `Status: ${message}`;
            statusMessage.className = `mt-4 p-3 rounded-lg ${color} border transition-all duration-500`;
            statusMessage.classList.remove('hidden');
        }


        // --- Socket.IO Event Handlers ---
        socket.on('connect', () => {
            logMessage(`Connected to WebSocket server at ${SOCKET_URL}`, 'info');
            setStatus('Ready to configure and start listening.', 'info');
        });

        socket.on('disconnect', () => {
            isListening = false;
            startListeningBtn.disabled = false;
            stopListeningBtn.disabled = true;
            setStatus('Disconnected from the server. Check backend status.', 'error');
        });

        // Receive status updates from the backend (success/failure messages)
        socket.on('status', ({ type, message }) => {
            setStatus(message, type);
            if (type === 'success') {
                isListening = true;
                startListeningBtn.disabled = true;
                stopListeningBtn.disabled = false;
            } else if (type === 'error') {
                isListening = false;
                startListeningBtn.disabled = false;
                stopListeningBtn.disabled = true;
            }
        });

        // Receive the actual log data from the backend
        socket.on('newLog', (logData) => {
            logMessage(`New Log for ${logData.name}`, 'log', logData);
        });

        // --- Frontend Button Actions ---
        function startListening() {
            if (isListening) return;

            const address = contractAddressInput.value.trim();
            const eventFragment = eventNameInput.value.trim();
            const abi = abiInput.value.trim();

            if (!address || !eventFragment || !abi) {
                setStatus("Please fill in all fields (Address, Event, and ABI).", 'error');
                return;
            }
            
            setStatus("Sending configuration to server...", 'info');
            
            // Emit the configuration to the backend to start the listener
            socket.emit('startListening', { address, eventFragment, abi });
        }

        function stopListening() {
            if (!isListening) return;

            // Emit the stop command to the backend
            socket.emit('stopListening');

            // Update local state immediately
            isListening = false;
            startListeningBtn.disabled = false;
            stopListeningBtn.disabled = true;
            setStatus('Stopping listener...', 'info');
        }

        // Initialize connection status on load
        document.addEventListener('DOMContentLoaded', () => {
            setStatus('Attempting to connect to the server...', 'info');
        });

    </script>
